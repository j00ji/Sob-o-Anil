# Charismatic event chain ? Pillar 1: Speaking in Tongues
############################################################

character_event = {
	id = charismatic.1000
	title = charismatic.1000.t
	desc = charismatic.1000.d
	picture = GFX_evt_mystical_experience

	trigger = {
		religion = charismatic
		piety = 20
	}

	mean_time_to_happen = { months = 120 }

	option = {
		name = charismatic.1000.a
		ai_chance = { factor = 70 }
		add_piety = 25
		set_character_flag = charismatic_tongues_experience
		# schedule immediate followup roll for trait gains / risk
		trigger_event = { id = charismatic.1005 days = 1 }
	}

	option = {
		name = charismatic.1000.b
		ai_chance = { factor = 20 }
		# interpretation requires study
		condition = { learning = 12 }
		add_piety = 15
		set_character_flag = charismatic_tongues_interpreter
		trigger_event = { id = charismatic.1001 days = 1 }
	}

	option = {
		name = charismatic.1000.c
		ai_chance = { factor = 10 }
		add_piety = -5
		add_opinion = { target = ROOT value = 5 }
		add_prestige = -15
		set_character_flag = charismatic_tongues_restrained
	}
}

character_event = {
	# Followup: roll for gifts/risks after a tongues experience
	id = charismatic.1005
	hidden = yes
	trigger = { has_character_flag = charismatic_tongues_experience }
	mean_time_to_happen = { days = 1 }
	immediate = yes

	option = {
		name = charismatic.1005.a
		ai_chance = { factor = 100 }
		# Weighted outcomes: mystic most likely, poet occasional, possessed rare
		random_list = {
			70 = { gain_trait = mystic }
			20 = { gain_trait = poet }
			10 = { gain_trait = possessed }
		}
		# small additional piety for being visibly spirit-filled
		add_piety = 10
	}
}
