narrative_event = {
	id = camino.10
	title=camino.10.t
	desc=camino.10.d
	picture=GFX_observe_camino
	is_triggered_only=yes
  	immediate = {
		random_demesne_province = {
    		limit = { 
    			any_neighbor_province = {
    				NOT = {
    					owner = { character = ROOT } 
    				} 
    			} 
    		}
    		random_neighbor_province = {
    			limit = { 
    				NOT = {
    					owner = { 
    						character = ROOT 
    					} 
    				} 
    			}
    			county ={
    				ROOT ={
						save_persistent_event_target = { 
							name = prophesied_province
							scope = PREV
						}
					}
				}
    		}
    	}   
  	}
	option = {
		name = camino.10.a
		set_character_flag = camino_prophecy_current
		if = {
			limit = {
				NOT = {
					OR = {
						trait = adept_astronomer
						trait = master_astronomer
					}
				}
			}
			add_character_modifier = {
	    		name = camino_conquest_prophecy
    			years = 2
			}
		}
		else_if = {
			limit = {
				trait = adept_astronomer
			}
			add_character_modifier = {
	    		name = camino_conquest_prophecy
    			years = 3			
			}
		}
		else = {
			add_character_modifier = {
	    		name = camino_conquest_prophecy
    			years = 4			
			}
		}
		persistent_event_target:prophesied_province={
			add_claim = ROOT
		}
	}
}

	id = camino.11
	title=camino.11.t
	desc=camino.11.d
	religion = camino_estelar
	trigger = {
		ROOT = {
			has_character_modifier = camino_conquest_prophecy
		}
		persistent_event_target:prophesied_province={
			holder = ROOT
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = camino.11.a
		piety = 50
		prestige = 50
		clr_character_flag = camino_prophecy_current
		remove_character_modifier = camino_conquest_prophecy
		if = {
			limit = {
				NOT = {
					OR = {
						trait = initiate_astronomer
						trait = adept_astronomer
						trait = master_astronomer
					}
				}				
			}
			add_trait = initiate_astronomer
		}
		else_if = {
			limit = {
				trait = initiate_astronomer				
			}
			random = {
				chance = 33
	    		mult_modifier = {
					factor = 1.2
					trait = zealous
    			}
				remove_trait = initiate_astronomer
				add_trait = adept_astronomer
			}
		}
		else_if = {
			limit = {
				trait = adept_astronomer				
			}
			random = {
				chance = 33
	    		mult_modifier = {
					factor = 1.2
					trait = zealous
    			}
				remove_trait = adept_astronomer
				add_trait = master_astronomer
			}
		}
		if = {
			limit = {
				NOT = {
					trait = zealous
				}
			}
			random = {
				chance = 20
				add_trait = zealous
			}
		}
		clear_persistent_event_target = prophesied_province
	}
}
