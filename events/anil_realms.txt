namespace = SOAREALM

#Renaming Kingdoms

# Special event for Fortaleza to be renamed to Ceará
character_event = {
	id = SOAREALM.1
	title = SOAREALM.1.T
	desc = SOAREALM.1.D
	
	picture = GFX_evt_council
	border = GFX_event_normal_frame_diplomacy
	
	only_playable = yes
	
	trigger = {
		has_landed_title = k_ceara
		NOT = { dynasty = 1 } # Belchioringian dynasty
		NOT = { has_global_flag = fortaleza_renamed }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = EVTOPTA62930
		set_global_flag = fortaleza_renamed
		k_ceara = {
			custom_tooltip = { text = tooltip_fortaleza_to_ceara }
			hidden_tooltip = {
				set_name=""
				adjective=""
			}
		}
	}
}
